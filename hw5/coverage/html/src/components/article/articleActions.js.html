export const fetchArticle = () => {
		return (dispatch) => {
		resource('GET', 'articles').then((r) => {
		const sortedArticles = r.articles.sort(function(a, b) {
            let x = new Date(a.date)
            let y = new Date(b.date)
            return ((x>y) ? -1 :((x<y)?1:0))
        });
        dispatch({
        	type: FETCH_ARTICLES,
        	sortedArticles: sortedArticles
        })
	})}
}
